Zoomer Script 

bind MWHEELDOWN "min"
bind MWHEELUP "zoom_60"
alias zoom_90 "sensitivity 3;fov 90;bind MWHEELUP zoom_60;bind MWHEELDOWN min;alias zoom zoom_90"
alias zoom_60 "sensitivity 2.5;fov 60;bind MWHEELUP zoom_40;bind MWHEELDOWN zoom_90;alias zoom zoom_60"
alias zoom_40 "sensitivity 2;fov 40;bind MWHEELUP zoom_30;bind MWHEELDOWN zoom_60;alias zoom zoom_40"
alias zoom_30 "sensitivity 1.5;fov 30;bind MWHEELUP zoom_20;bind MWHEELDOWN zoom_40;alias zoom zoom_30"
alias zoom_20 "sensitivity 1;fov 20;bind MWHEELUP max;bind MWHEELDOWN zoom_30;alias zoom zoom_20"
alias +invcyc "bind MWHEELUP next;bind MWHEELDOWN prev;alias uncyc invuse"
alias min "echo Minimum zoom;unbind MWHEELDOWN"
alias max "echo Maximum zoom;unbind MWHEELUP"
